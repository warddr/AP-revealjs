def fibonacci(n):
	"""Return Fibonacci sequence up to n."""
	result = []
	a, b = 0, 1
	while a < n:
		result.append(a)
		a, b = b, a+b
	return result

if __name__ == "__main__":
	num = int(input("Enter a number: "))
	fib_sequence = fibonacci(num)
	print(f"Fibonacci sequence up to {num}: {fib_sequence}")
